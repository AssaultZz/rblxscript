	LOOP_DELAY            = 5,                -- Seconds between each full loop

-- ============================================================
--  HELPER: Debug logger
-- ============================================================

local function debugLog(msg)
	if CONFIG.DEBUG then
		print("[CaseParadise] " .. tostring(msg))
	end
end

-- ============================================================
--  HELPER: Find a remote by name in ReplicatedStorage
-- ============================================================

local function findRemote(name)
	local remote = ReplicatedStorage:FindFirstChild(name, true)
	if not remote then
		debugLog("WARNING: Remote not found: " .. name)
	end
	return remote
end

-- ============================================================
--  HELPER: Safely fire a RemoteEvent
-- ============================================================

local function safeFireRemote(remote, ...)
	if not remote then return false end
	local ok, err = pcall(function() remote:FireServer(...) end)
	if not ok then debugLog("FireRemote error: " .. tostring(err)) end
	return ok
end

-- ============================================================
--  HELPER: Safely invoke a RemoteFunction
-- ============================================================

local function safeInvokeRemote(remote, ...)
	if not remote then return nil end
	local ok, result = pcall(function() return remote:InvokeServer(...) end)
	return ok and result or nil
end

-- ============================================================
--  CORE: Open a case by name
-- ============================================================

local function openCase(caseName)
	debugLog("Opening case: " .. caseName)
	local remote = findRemote("OpenCase") -- ← update name if needed
	return safeFireRemote(remote, caseName)
end

-- ============================================================
--  CORE: Get inventory from server
-- ============================================================

local function getInventory()
	local remote = findRemote("GetInventory") -- ← update name if needed
	return safeInvokeRemote(remote)
end

-- ============================================================
--  CORE: Sell one item
-- ============================================================

local function sellItem(item)
	local remote = findRemote("SellItem") -- ← update name if needed
	local id = item.Id or item.UUID or item.Name or item
	return safeFireRemote(remote, id)
end

-- ============================================================
--  CORE: Sell all items below threshold
-- ============================================================

local function sellLowValueItems()
	debugLog("Selling items below value: " .. CONFIG.KEEP_VALUE_THRESHOLD)
	local inventory = getInventory()
	if not inventory then
		debugLog("No inventory returned.")
		return 0, 0
	end

	local sold, kept = 0, 0
	for _, item in ipairs(inventory) do
		local val = item.Value or item.Price or item.Worth or 0
		if val < CONFIG.KEEP_VALUE_THRESHOLD then
			if sellItem(item) then sold += 1 end
			task.wait(0.1)
		else
			kept += 1
		end
	end
	debugLog("Sold: " .. sold .. " | Kept: " .. kept)
	return sold, kept
end

-- ============================================================
--  GUI LOGIC: Update status display
-- ============================================================

local function setStatus(msg, running)
	StatusLabel.Text = "Status: " .. msg

	-- Green dot when running, grey when idle
	if running then
		StatusDot.BackgroundColor3 = Color3.fromRGB(80, 220, 120)
	else
		StatusDot.BackgroundColor3 = Color3.fromRGB(100, 100, 130)
	end
end

-- ============================================================
--  GUI LOGIC: Read config values from text boxes
--  Validates numbers and falls back to defaults if invalid
-- ============================================================

local function applyConfigFromGUI()
	-- Case name (just use whatever string is typed)
	local name = CaseNameBox.Text
	if name and name ~= "" then
		CONFIG.CASE_NAME = name
	end

	-- Sell delay — must be a positive number
	local delay = tonumber(SellDelayBox.Text)
	if delay and delay > 0 then
		CONFIG.SELL_DELAY = delay
	else
		SellDelayBox.Text = tostring(CONFIG.SELL_DELAY) -- reset to last valid
		debugLog("Invalid sell delay — keeping: " .. CONFIG.SELL_DELAY)
	end

	-- Keep value threshold — must be a non-negative number
	local threshold = tonumber(ThresholdBox.Text)
	if threshold and threshold >= 0 then
		CONFIG.KEEP_VALUE_THRESHOLD = threshold
	else
		ThresholdBox.Text = tostring(CONFIG.KEEP_VALUE_THRESHOLD)
		debugLog("Invalid threshold — keeping: " .. CONFIG.KEEP_VALUE_THRESHOLD)
	end

	debugLog("Config applied → Case: " .. CONFIG.CASE_NAME ..
		" | Delay: " .. CONFIG.SELL_DELAY ..
		" | Threshold: " .. CONFIG.KEEP_VALUE_THRESHOLD)
end

-- ============================================================
--  CORE: Main auto-loop (runs in its own task thread)
-- ============================================================

local function runLoop()
	while isRunning do
		-- Read the latest GUI values each loop
		applyConfigFromGUI()

		-- STEP 1: Open the case
		setStatus("Opening: " .. CONFIG.CASE_NAME, true)
		debugLog("Opening case: " .. CONFIG.CASE_NAME)
		openCase(CONFIG.CASE_NAME)

		-- STEP 2: Countdown while waiting to sell
		local elapsed = 0
		while elapsed < CONFIG.SELL_DELAY and isRunning do
			local remaining = CONFIG.SELL_DELAY - elapsed
			setStatus("Selling in " .. math.ceil(remaining) .. "s…", true)
			task.wait(1)
			elapsed += 1
		end

		-- If stopped mid-wait, exit cleanly
		if not isRunning then break end

		-- STEP 3: Sell low-value items
		setStatus("Selling items…", true)
		local sold, kept = sellLowValueItems()
		setStatus("Sold " .. sold .. " | Kept " .. kept, true)
		task.wait(2) -- brief pause so user can see the result

		-- STEP 4: Short pause before next case
		if isRunning then
			setStatus("Next case in " .. CONFIG.LOOP_DELAY .. "s…", true)
			task.wait(CONFIG.LOOP_DELAY)
		end
	end

	-- Loop ended (user pressed Stop)
	setStatus("Idle", false)
	debugLog("Loop stopped.")
end

-- ============================================================
--  GUI LOGIC: Start / Stop button
-- ============================================================

StartBtn.MouseButton1Click:Connect(function()
	if not isRunning then
		-- ── START ──
		isRunning = true

		-- Update button appearance to show "Stop" state
		StartBtn.Text             = "■  STOP"
		StartBtn.BackgroundColor3 = Color3.fromRGB(180, 40, 60)
		BtnGrad.Color = ColorSequence.new({
			ColorSequenceKeypoint.new(0, Color3.fromRGB(210, 55, 75)),
			ColorSequenceKeypoint.new(1, Color3.fromRGB(150, 30, 50)),
		})

		setStatus("Starting…", true)

		-- Spin up the loop in a separate task (so GUI stays responsive)
		loopThread = task.spawn(runLoop)

	else
		-- ── STOP ──
		isRunning = false

		-- Update button back to "Start" state
		StartBtn.Text             = "▶  START"
		StartBtn.BackgroundColor3 = Color3.fromRGB(90, 55, 200)
		BtnGrad.Color = ColorSequence.new({
			ColorSequenceKeypoint.new(0, Color3.fromRGB(110, 65, 230)),
			ColorSequenceKeypoint.new(1, Color3.fromRGB(70,  40, 170)),
		})

		setStatus("Stopping…", false)
		debugLog("Stop requested by user.")
	end
end)

-- ============================================================
--  GUI LOGIC: Drag to move the window
-- ============================================================

local dragging     = false
local dragStart    = nil   -- mouse position when drag began
local startPos     = nil   -- window position when drag began

TitleBar.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or
		input.UserInputType == Enum.UserInputType.Touch then
		dragging  = true
		dragStart = input.Position
		startPos  = Window.Position
	end
end)

TitleBar.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or
		input.UserInputType == Enum.UserInputType.Touch then
		dragging = false
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if dragging and (
		input.UserInputType == Enum.UserInputType.MouseMovement or
			input.UserInputType == Enum.UserInputType.Touch
		) then
		local delta = input.Position - dragStart
		Window.Position = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
	end
end)
