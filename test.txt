-- ============================================================

local function debugLog(msg)
	if CONFIG.DEBUG then print("[CaseParadise] " .. tostring(msg)) end
end

local function findRemote(name)
	local remote = ReplicatedStorage:FindFirstChild(name, true)
	if not remote then debugLog("WARNING: Remote not found: " .. name) end
	return remote
end

local function safeFireRemote(remote, arg1)
	if not remote then return false end
	local ok, err = pcall(function() remote:FireServer(arg1) end)
	if not ok then debugLog("FireRemote error: " .. tostring(err)) end
	return ok
end

local function safeInvokeRemote(remote, arg1)
	if not remote then return nil end
	local ok, result = pcall(function() return remote:InvokeServer(arg1) end)
	return ok and result or nil
end

-- ============================================================
--  CORE FUNCTIONS
-- ============================================================

local function openCase(caseName)
	debugLog("Opening case: " .. caseName)
	return safeFireRemote(findRemote("OpenCase"), caseName)
end

local function getInventory()
	return safeInvokeRemote(findRemote("GetInventory"))
end

local function sellItem(item)
	local id = item.Id or item.UUID or item.Name or item
	return safeFireRemote(findRemote("SellItem"), id)
end

local function sellLowValueItems()
	local inventory = getInventory()
	if not inventory then return 0, 0 end
	local sold, kept = 0, 0
	for _, item in ipairs(inventory) do
		local val = item.Value or item.Price or item.Worth or 0
		if val < CONFIG.KEEP_VALUE_THRESHOLD then
			if sellItem(item) then sold = sold + 1 end
			wait(0.1)
		else
			kept = kept + 1
		end
	end
	return sold, kept
end

local function setStatus(msg, running)
	StatusLabel.Text = "Status: " .. msg
	StatusDot.BackgroundColor3 = running
		and Color3.fromRGB(80, 220, 120)
		or  Color3.fromRGB(100, 100, 130)
end

local function applyConfigFromGUI()
	local delay = tonumber(SellDelayBox.Text)
	if delay and delay > 0 then CONFIG.SELL_DELAY = delay
	else SellDelayBox.Text = tostring(CONFIG.SELL_DELAY) end

	local threshold = tonumber(ThresholdBox.Text)
	if threshold and threshold >= 0 then CONFIG.KEEP_VALUE_THRESHOLD = threshold
	else ThresholdBox.Text = tostring(CONFIG.KEEP_VALUE_THRESHOLD) end
end

local function runLoop()
	while isRunning do
		applyConfigFromGUI()

		local caseName = CASES[CONFIG.CASE_INDEX].name

		-- Open however many cases the user selected
		for i = 1, CONFIG.OPEN_COUNT do
			if not isRunning then break end
			setStatus("Opening " .. i .. "/" .. CONFIG.OPEN_COUNT .. ": " .. caseName, true)
			openCase(caseName)
			wait(0.5) -- small gap between opens
		end

		-- Countdown to sell
		local elapsed = 0
		while elapsed < CONFIG.SELL_DELAY and isRunning do
			setStatus("Selling in " .. math.ceil(CONFIG.SELL_DELAY - elapsed) .. "s", true)
			wait(1)
			elapsed = elapsed + 1
		end

		if not isRunning then break end

		setStatus("Selling items...", true)
		local sold, kept = sellLowValueItems()
		setStatus("Sold " .. sold .. " | Kept " .. kept, true)
		wait(2)

		if isRunning then
			setStatus("Next loop in " .. CONFIG.LOOP_DELAY .. "s", true)
			wait(CONFIG.LOOP_DELAY)
		end
	end
	setStatus("Idle", false)
end

StartBtn.MouseButton1Click:Connect(function()
	if not isRunning then
		isRunning = true
		StartBtn.Text = "STOP"
		StartBtn.BackgroundColor3 = Color3.fromRGB(180, 40, 60)
		BtnGrad.Color = ColorSequence.new({
			ColorSequenceKeypoint.new(0, Color3.fromRGB(210, 55, 75)),
			ColorSequenceKeypoint.new(1, Color3.fromRGB(150, 30, 50)),
		})
		setStatus("Starting...", true)
		loopThread = spawn(runLoop)
	else
		isRunning = false
		StartBtn.Text = "START"
		StartBtn.BackgroundColor3 = Color3.fromRGB(90, 55, 200)
		BtnGrad.Color = ColorSequence.new({
			ColorSequenceKeypoint.new(0, Color3.fromRGB(110, 65, 230)),
			ColorSequenceKeypoint.new(1, Color3.fromRGB(70, 40, 170)),
		})
		setStatus("Stopping...", false)
	end
end)
